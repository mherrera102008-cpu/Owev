```typescript
// Next.js 16 route protection — replaces deprecated middleware.ts
// All dashboard routes are protected; auth routes and webhooks are public.
// Source: https://nextjs.org/docs/app/api-reference/file-conventions/proxy
//         https://clerk.com/docs/nextjs/getting-started/quickstart
import { clerkMiddleware, createRouteMatcher } from '@clerk/nextjs/server';

const isPublicRoute = createRouteMatcher([
  '/',                         // Root redirect page
  '/sign-in(.*)',              // Clerk hosted sign-in (catch-all for Clerk's sub-routes)
  '/sign-up(.*)',              // Clerk hosted sign-up (catch-all for Clerk's sub-routes)
  '/api/webhooks(.*)',         // Stripe + Clerk webhooks — MUST be public for external services
]);

export default clerkMiddleware(async (auth, request) => {
  if (!isPublicRoute(request)) {
    await auth.protect();     // Redirects to /sign-in if no valid Clerk session
  }
});

export const config = {
  matcher: [
    // Skip Next.js internals and all static files (.ico, .png, .css, .js, etc.)
    '/((?!_next|[^?]*\\.(?:html?|css|js(?!on)|jpe?g|webp|png|gif|svg|ttf|woff2?|ico|csv|docx?|xlsx?|zip|webmanifest)).*)',
    '/(api|trpc)(.*)',
  ],
};
```

```typescript
// convex/http.ts
// Clerk webhook handler — receives user.created and user.updated events from Clerk.
// Verifies Svix signature before trusting any payload.
// On success: upserts tenant document via convex/users.ts internalMutation.
//
// IMPORTANT: This endpoint is served at:
//   https://<your-deployment>.convex.site/clerk-users-webhook
//   ↑ Note: .convex.SITE — NOT .convex.cloud
//   The .site domain is Convex's HTTP action subdomain (separate from WebSocket domain).
//
// Clerk Dashboard configuration required:
//   Webhooks → Add endpoint → URL: https://<deployment>.convex.site/clerk-users-webhook
//   Subscribe to: user.created, user.updated
//   Copy the Signing Secret → set as CLERK_WEBHOOK_SECRET in Convex dashboard env vars
//
// Sources:
//   .planning/phases/01-foundation/01-RESEARCH.md Pattern 6
//   .planning/research/PITFALLS.md (Clerk webhook pitfalls)
import { httpRouter } from "convex/server";
import { httpAction } from "./_generated/server";
import { internal } from "./_generated/api";
import type { WebhookEvent } from "@clerk/backend";
import { Webhook } from "svix";

const http = httpRouter();

http.route({
  path: "/clerk-users-webhook",
  method: "POST",
  handler: httpAction(async (ctx, request) => {
    const webhookSecret = process.env.CLERK_WEBHOOK_SECRET;
    if (!webhookSecret) {
      console.error("CLERK_WEBHOOK_SECRET not set in Convex environment variables");
      return new Response("Server configuration error", { status: 500 });
    }

    // Extract Svix signature headers — all three required for verification
    const svixId = request.headers.get("svix-id");
    const svixTimestamp = request.headers.get("svix-timestamp");
    const svixSignature = request.headers.get("svix-signature");

    if (!svixId || !svixTimestamp || !svixSignature) {
      return new Response("Missing required Svix headers", { status: 400 });
    }

    // Read raw body as text — do NOT use request.json() before verification
    // (json() consumes the stream; Svix needs the raw bytes for HMAC verification)
    const payloadString = await request.text();
    const wh = new Webhook(webhookSecret);
    let event: WebhookEvent;

    try {
      event = wh.verify(payloadString, {
        "svix-id": svixId,
        "svix-timestamp": svixTimestamp,
        "svix-signature": svixSignature,
      }) as WebhookEvent;
    } catch (err) {
      console.error("Svix signature verification failed:", err);
      return new Response("Invalid webhook signature", { status: 400 });
    }

    // Only handle user lifecycle events — ignore everything else
    switch (event.type) {
      case "user.created":
      case "user.updated":
        await ctx.runMutation(internal.users.upsertFromClerk, {
          clerkId: event.data.id,
          email: event.data.email_addresses[0]?.email_address,
        });
        break;
      default:
        // Acknowledge but do not process other event types
        break;
    }

    // Always return 200 for recognized webhook deliveries to prevent Clerk retries
    return new Response(null, { status: 200 });
  }),
});

export default http;
```
