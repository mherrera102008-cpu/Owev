```typescript
// convex/schema.ts
// All table definitions with indexes for phases 1-4.
// CRITICAL: Indexes must exist before any data is written — no migration needed if defined first.
// Source: .planning/phases/01-foundation/01-RESEARCH.md Pattern 4
import { defineSchema, defineTable } from "convex/server";
import { v } from "convex/values";

export default defineSchema({
  // -------------------------------------------------------------------------
  // TENANTS — one row per user, created by Clerk webhook on first signup.
  // ownerId = Clerk userId (identity.subject) — this IS the tenant key for v1.
  // by_stripe_customer index required by Phase 4: Stripe webhooks arrive with
  // customerId, not ownerId — without this index, webhook→tenant lookup is O(n).
  // -------------------------------------------------------------------------
  tenants: defineTable({
    ownerId: v.string(),                   // Clerk userId (identity.subject)
    clerkId: v.string(),                   // Same as ownerId; explicit alias for clarity
    email: v.optional(v.string()),
    stripeCustomerId: v.optional(v.string()),
    stripeSubscriptionId: v.optional(v.string()),
    subscriptionStatus: v.union(
      v.literal("active"),
      v.literal("trialing"),
      v.literal("past_due"),
      v.literal("canceled"),
      v.literal("none")
    ),
    reminderConfig: v.object({
      daysBefore: v.array(v.number()),     // e.g. [7, 3, 1] — days before due date
      daysAfter: v.array(v.number()),      // e.g. [1, 3, 7] — days after due date
    }),
    createdAt: v.number(),                 // Unix timestamp ms
  })
    .index("by_owner", ["ownerId"])
    .index("by_stripe_customer", ["stripeCustomerId"]),

  // -------------------------------------------------------------------------
  // INVOICES — core data model. Data written in Phase 2.
  // Status union: upcoming | due | overdue | paid
  //   - upcoming: dueDate > today (daily cron does not touch these)
  //   - due:      dueDate == today (daily cron transitions from upcoming)
  //   - overdue:  dueDate < today  (daily cron transitions from due)
  //   - paid:     terminal state, set by user action
  // scheduledReminderIds: stored so Phase 3 can cancel reminders on mark-paid.
  // by_owner_status: compound index for Phase 3 reminder dispatch query
  //   (filter by ownerId + status != "paid" in one indexed scan).
  // -------------------------------------------------------------------------
  invoices: defineTable({
    ownerId: v.string(),                   // Clerk userId — tenant key
    clientName: v.string(),
    clientEmail: v.string(),
    amount: v.number(),                    // Stored in cents (integer) to avoid float rounding
    dueDate: v.number(),                   // Unix timestamp ms (start of day UTC)
    status: v.union(
      v.literal("upcoming"),
      v.literal("due"),
      v.literal("overdue"),
      v.literal("paid")
    ),
    description: v.optional(v.string()),
    invoiceNumber: v.optional(v.number()), // Auto-incremented per tenant — assigned in Phase 2
    scheduledReminderIds: v.optional(
      v.array(v.id("_scheduled_functions"))
    ),                                     // Phase 3: job IDs for cancellation on mark-paid
    createdAt: v.number(),                 // Unix timestamp ms
  })
    .index("by_owner", ["ownerId"])
    .index("by_owner_status", ["ownerId", "status"]),

  // -------------------------------------------------------------------------
  // REMINDER LOG — audit trail of every email send attempt (Phase 3).
  // Defined in schema now so the table exists; data written in Phase 3.
  // Used for: idempotency guard, send history UI, debugging delivery failures.
  // -------------------------------------------------------------------------
  reminderLog: defineTable({
    invoiceId: v.id("invoices"),
    ownerId: v.string(),
    sentAt: v.number(),                    // Unix timestamp ms
    reminderType: v.string(),              // e.g. "7_days_before", "on_due_date", "3_days_after"
    status: v.union(
      v.literal("sent"),
      v.literal("failed")
    ),
    resendMessageId: v.optional(v.string()),
    errorMessage: v.optional(v.string()),
  })
    .index("by_invoice", ["invoiceId"])
    .index("by_owner", ["ownerId"]),

  // -------------------------------------------------------------------------
  // PROCESSED WEBHOOKS — idempotency table for Stripe events (Phase 4).
  // Defined now so the index exists before Phase 4 writes to it.
  // Pattern: check before process, insert after — entire operation in one mutation.
  // -------------------------------------------------------------------------
  processedWebhooks: defineTable({
    stripeEventId: v.string(),             // Stripe event ID (e.g. "evt_...")
    processedAt: v.number(),               // Unix timestamp ms
  })
    .index("by_stripe_event_id", ["stripeEventId"]),
});
```

```typescript
// convex/users.ts
// Two exports:
//   1. getCurrentUserId — canonical auth guard used in every tenant-scoped query/mutation
//   2. upsertFromClerk — internalMutation called by Clerk webhook (not callable from client)
//
// SECURITY RULE: Never accept userId as a client argument.
//   Always use ctx.auth.getUserIdentity() server-side.
//   identity.subject = Clerk userId = the tenant key.
//
// Sources:
//   .planning/phases/01-foundation/01-RESEARCH.md Pattern 5 + Pattern 6
//   .planning/research/PITFALLS.md Pitfall 2 (client-supplied userId)
import { MutationCtx, QueryCtx, internalMutation } from "./_generated/server";
import { v } from "convex/values";

/**
 * Returns the Clerk userId (identity.subject) for the authenticated caller.
 * Throws if no valid Clerk JWT is present — never degrades silently.
 *
 * Usage: Call as the FIRST line of every tenant-scoped query and mutation.
 *
 * @example
 *   export const list = query({
 *     args: {},
 *     handler: async (ctx) => {
 *       const ownerId = await getCurrentUserId(ctx);
 *       return ctx.db.query("invoices")
 *         .withIndex("by_owner", q => q.eq("ownerId", ownerId))
 *         .collect();
 *     },
 *   });
 */
export async function getCurrentUserId(
  ctx: QueryCtx | MutationCtx
): Promise<string> {
  const identity = await ctx.auth.getUserIdentity();
  if (!identity) {
    throw new Error("Unauthenticated: valid Clerk session required");
  }
  // identity.subject = Clerk userId string (e.g. "user_2abc...")
  // This is the tenant key — all data is scoped to this value.
  return identity.subject;
}

/**
 * Creates or updates the tenant row when a user signs up or updates in Clerk.
 * Called exclusively from convex/http.ts Clerk webhook handler.
 *
 * SECURITY: internalMutation — cannot be called from the browser client.
 * This means the only path to create a tenant is via a verified Clerk webhook.
 *
 * Default reminderConfig: 7/3/1 days before, 1/3/7 days after due date.
 * These defaults are configurable per tenant in Phase 3.
 */
export const upsertFromClerk = internalMutation({
  args: {
    clerkId: v.string(),
    email: v.optional(v.string()),
  },
  handler: async (ctx, { clerkId, email }) => {
    // Check if tenant already exists — upsert to handle user.updated events too
    const existing = await ctx.db
      .query("tenants")
      .withIndex("by_owner", (q) => q.eq("ownerId", clerkId))
      .first();

    if (!existing) {
      // First login — create tenant with defaults
      await ctx.db.insert("tenants", {
        ownerId: clerkId,
        clerkId,
        email,
        subscriptionStatus: "none",  // No subscription until Phase 4
        reminderConfig: {
          daysBefore: [7, 3, 1],     // Send reminders 7, 3, and 1 day before due date
          daysAfter: [1, 3, 7],      // Send reminders 1, 3, and 7 days after due date
        },
        createdAt: Date.now(),
      });
    } else if (email && existing.email !== email) {
      // Sync email updates (e.g., user changed email in Clerk)
      await ctx.db.patch(existing._id, { email });
    }
    // If neither condition: no-op (idempotent)
  },
});
```
